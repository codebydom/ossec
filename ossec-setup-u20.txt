OSSEC SET UP

1.) sudo apt update && sudo apt upgrade
2.) sudo apt install -y wget unzip make gcc build-essential
3.) sudo apt install apache2
4.) sudo systemctl enable apache2 && sudo systemctl start apache2
5.) sudo apt install -y php php-cli php-common libapache2-mod-php apache2-utils sendmail inotify-tools
6.) sudo apt install libz-dev libssl-dev libpcre2-dev build-essential libsystemd-dev
7.) sudo apt install mailutils
8.) sudo sh install.sh
9.) enter in root@localhost for email
10.) sudo /var/ossec/bin/ossec-control start
11.) cd /srv/
12.) sudo git clone https://github.com/ossec/ossec-wui.git
13.) sudo mv /srv/ossec-wui /var/www/html
14.) cd /var/www/html/ossec-wui && sudo ./setup.sh
15.) enter www-data
16.) cd ~
17.) sudo chown -R www-data:www-data /var/www/html/ossec-wui/
18.) sudo chmod -R 755 /var/www/html/ossec-wui/
19.) sudo a2enmod rewrite
20.) sudo systemctl restart apache2
21.) xdg-open http://127.0.0.1/ossec-wui
