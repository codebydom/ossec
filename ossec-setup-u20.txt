OSSEC SET UP

1.) sudo apt update && sudo apt upgrade
2.) sudo apt install -y wget unzip make gcc build-essential
3.) sudo apt install apache2
4.) sudo systemctl enable apache2 && sudo systemctl start apache2
5.) sudo apt install -y php php-cli php-common libapache2-mod-php apache2-utils sendmail inotify-tools
6.) sudo apt install libz-dev libssl-dev libpcre2-dev build-essential libsystemd-dev
7.) sudo apt install mailutils
8.) wget https://github.com/ossec/ossec-hids/archive/refs/heads/master.zip && unzip master.zip
9.) rm master.zip && cd ./ossec-hids-master
10.) sudo sh install.sh
11.) enter in root@localhost for email
12.) sudo /var/ossec/bin/ossec-control start
13.) cd /srv/
14.) sudo git clone https://github.com/ossec/ossec-wui.git
15.) sudo mv /srv/ossec-wui /var/www/html
16.) cd /var/www/html/ossec-wui && sudo ./setup.sh
17.) enter www-data
18.) cd ~
19.) sudo chown -R www-data:www-data /var/www/html/ossec-wui/
20.) sudo chmod -R 755 /var/www/html/ossec-wui/
21.) sudo a2enmod rewrite
22.) sudo systemctl restart apache2
23.) xdg-open http://127.0.0.1/ossec-wui
